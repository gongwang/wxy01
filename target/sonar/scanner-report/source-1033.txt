jQuery(function($){
    // 备份jquery的ajax方法  
    var _ajax=$.ajax;
    // 重写ajax方法，先判断登录在执行success函数 
    $.ajax=function(opt){
        var _opt = $.extend(opt, {
			error:function(xhr){
				if(xhr.responseText.indexOf('用户登录')>0){
					window.location = '/login'
				}
			}
        });
        _ajax(_opt);
    };
});
